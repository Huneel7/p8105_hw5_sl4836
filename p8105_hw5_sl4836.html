fill_NA <- function(v) {
  
  numeric_result = 
    v %>%  
    mutate_if(is.numeric, 
              funs(ifelse(is.na(.),  mean(., na.rm = TRUE), .))) %>%
    select_if(is.numeric)
                    
  character_result = 
    v %>% 
    mutate_if(is.character, replace_na, replace = "virginica") %>% 
    select_if(is.character)
  
  tibble(numeric_result, character_result)
}
